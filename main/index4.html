window.addEventListener("mousemove", function (e) {
    document.querySelectorAll(".move").forEach(elem => {
        const {
            x,
            y
        } = e;
        const rect = elem.getBoundingClientRect();
        const mouseX = (x - rect.left - rect.width / 2) / 20;
        const mouseY = (y - rect.top - rect.height / 2) / 20;
        elem.style.textShadow = 
               `${mouseX}px ${-mouseY}px 0px rgba(255, 255, 255, 0.3), 
                ${-mouseX}px ${mouseY}px 0px rgba(0, 105, 92, 1), 
                ${mouseY}px ${-mouseX}px 0px rgba(183, 28, 28, 0.9), 
                ${-mouseY}px ${mouseX}px 0px rgba(105, 240, 174, 0.5)`; 
        })
    });

Image Slider

자바스크립트를 이용하여 좌, 우 버튼을 누르면 이미지들이 이동하는 슬라이드를 만들었습니다.

const sliderWrap = $(".slider-wrap");
    const sliderImg = $(".slider-img");
    const sliders = $(".sliders");
    const sliderBtn = $(".slider-btn");

    let slideCount = sliders.length;
    let slideWidth = sliders.width();
    let slideHeight = sliders.height();
    let slideTotalWidth = slideCount * slideWidth;

    sliders.css({
        width: slideWidth,
        height: slideHeight
    });
    sliderImg.css({
        width: slideTotalWidth,
        height: slideHeight
    });
    sliderImg.css({
        marginLeft: -slideWidth
    });
    $(".sliders:last-child").prependTo(sliderImg);

    function moveLeft() {
        sliderImg.animate({
            left: +slideWidth
        }, 400, function () {
            $(".sliders:last-child").prependTo(sliderImg);
            sliderImg.css("left", "");
        });
    }

    function moveRight() {
        sliderImg.animate({
            left: -slideWidth
        }, 400, function () {
            $(".sliders:first-child").appendTo(sliderImg);
            sliderImg.css("left", "");
        });
    }

    sliderBtn.find(".prev").click(function () {
        moveLeft()
    });
    sliderBtn.find(".next").click(function () {
        moveRight()
    });

Twinkle stars

HTML <Canvas>요소를 이용하였고, 랜덤값을 이용하여 별이 반짝이는 효과를 만들었습니다.

var canvas;
var context;
var screenH;
var screenW;
var stars = [];
var fps = 50;
var numStars = 1000;

$('document').ready(function () {

    //화면크기 계산하기
    screenH = $(window).height();
    screenW = $(window).width();

    //캔버스 가져오기
    canvas = $('#wrapBox');

    //캔버스 채우기
    canvas.attr('height', screenH);
    canvas.attr('width', screenW);
    context = canvas[0].getContext('2d');

    //별 만들기
    for (var i = 0; i < numStars; i++) {
        var x = Math.round(Math.random() * screenW);
        var y = Math.round(Math.random() * screenH);
        var length = 1 + Math.random() * 2;
        var opacity = Math.random();

        //새로운 별 만들고 그리기
        var star = new Star(x, y, length, opacity);

        //배열 추가
        stars.push(star);
    }
    setInterval(animate, 100 / fps);
});

function animate() {
    context.clearRect(0, 0, screenW, screenH);
    $.each(stars, function () {
        this.draw(context);
    })
}
function Star(x, y, length, opacity) {
    this.x = parseInt(x);
    this.y = parseInt(y);
    this.length = parseInt(length);
    this.opacity = opacity;
    this.factor = 1;
    this.increment = Math.random() * .03;
}

Star.prototype.draw = function () {
    context.rotate((Math.PI * 1 / 10));

    context.save();

    context.translate(this.x, this.y);

    //투명도 변경
    if (this.opacity > 1) {
        this.factor = -1;
    } else if (this.opacity <= 0) {
        this.factor = 1;

        this.x = Math.round(Math.random() * screenW);
        this.y = Math.round(Math.random() * screenH);
    }

    this.opacity += this.increment * this.factor;

    context.beginPath()
    for (var i = 5; i--;) {
        context.lineTo(0, this.length);
        context.translate(0, this.length);
        context.rotate((Math.PI * 2 / 10));
        context.lineTo(0, -this.length);
        context.translate(0, -this.length);
        context.rotate(-(Math.PI * 6 / 10));
    }
    context.lineTo(0, this.length);
    context.closePath();
    context.fillStyle = "rgba(255, 255, 200, " + this.opacity + ")";
    context.shadowBlur = 5;
    context.shadowColor = '#ffe433';
    context.fill();

    context.restore();
}
